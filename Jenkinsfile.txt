  agent {
    kubernetes {
      yaml """

kind: Pod
metadata:
  labels:
    some-label: some-label-value
spec:
  nodeSelector:
    openstack-control-plane: enabled
  containers:
  - name: maven
    image: maven:alpine
    command:
    - cat
    tty: true
  - name: docker-dind
    image: docker:dind
    command:
    - cat
    tty: true
"""
    }
  }
  stages {
    stage('Run maven') {
      steps {
        container('maven') {
          sh 'mvn -version'
        }
        container('busybox') {
          sh 'docker --version'
        }
      }
    }
  }
